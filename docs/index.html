5 分钟快速开始

# 1. 克隆
git clone https://github.com/yourname/myagent.git
cd myagent

# 2. 安装
bun install

# 3. 配置
cp .env.example .env
# 填入 ANTHROPIC_API_KEY 或 DEEPSEEK_API_KEY

# 4. 跑(paper 模式)
QUANT_MODE=paper bun run dev
